Usage
Import: 
import PopoverModule from '@ng1bs5/popover';

angular.module('myApp', [PopoverModule.name]);
HTML: 
<button bs-popover 
        data-bs-title="Title"
        data-bs-content="Content"
        data-bs-placement="top">
    Click me
</button>
